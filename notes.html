数组

                            var arr = new Array();
                            arr = [1,2,3,4];
                            arr.push(5); //在数组末端增加项
                            arr; // [1, 2, 3,4,5];
                        
                            arr.push(0); //在数组开始位置增加项
                            arr; // [0，1, 2, 3,4,5];
                        
                            arr.pop(); //在数组末端删除项
                            arr; // [0，1, 2, 3,4];
                        
                            arr.shift(); //在数组开始位置删除项
                            arr; // [1, 2, 3,4]
                        
                            arr.join("_"); // 1_2_3_4_5   将数组转换成字符串
                        
                            arr.reverse(); //[5, 4, 3, 2, 1]  将数组逆序,原数组被修改
                        
                            var arr = ["a", "d", "c", "b"];
                            arr.sort(); // ["a", "b", "c", "d"] 排序  将数组逆序,原数组被修改
                        
                            arr = [13, 24, 51, 3];
                            arr.sort(); // [13, 24, 3, 51]  将数字转换成字符串进行排序，将数组逆序,原数组被修改
                            arr; // [13, 24, 3, 51]
                        
                            arr.sort(function(a, b) {
                            return a - b;  //如果b-a,则为从大到小排列
                            }); // [3, 13, 24, 51]
                        
                            var arr = [1, 2, 3];
                            arr.concat(4, 5); // [1, 2, 3, 4, 5]   //数组合并   原数组未被修改
                            arr; // [1, 2, 3]

                            arr.concat([10, 11], 13); // [1, 2, 3, 10, 11, 13]  //数组会被拉平

                            arr.concat([1, [2, 3]]); // [1, 2, 3, 1, [2, 3]]   //数组只会被拉平一次

                        
                            var arr = [1, 2, 3, 4, 5];
                            arr.slice(1, 3); // [2, 3]   返回数组片段，第一位表示从第几位开始，第二位表示想要获取的最后一个元素的下一个索引
                            arr.slice(1); // [2, 3, 4, 5]
                            arr.slice(1, -1); // [2, 3, 4]
                            arr.slice(-4, -3); // [2]
                        
                            var arr = [1, 2, 3, 4, 5];
                            arr.splice(2); // returns [3, 4, 5]  数组拼接，原数组被修改
                            arr; // [1, 2];

                            arr = [1, 2, 3, 4, 5];
                            arr.splice(2, 2); // returns [3, 4]   //第一位表示从第几位开始，第二位表示需要删除的元素个数
                            arr; // [1, 2, 5];

                            arr = [1, 2, 3, 4, 5];
                            arr.splice(1, 1, 'a', 'b'); // returns [2]  第二位以后的表示想要在删除位置添加的元素
                            arr; // [1, "a", "b", 3, 4, 5]
